Banco de dados: `thecrows`
-- Tabela para armazenar informações sobre corvos
CREATE TABLE `Crow` (
    `CrowId` INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `QrCode` VARCHAR(50) NOT NULL,
    `Name` VARCHAR(150) NOT NULL,
    `Tip` VARCHAR(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

-- Tabela para armazenar fotografias dos corvos
CREATE TABLE `Photograph` (
    `PhotoId` INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `Photo` MEDIUMBLOB NOT NULL,
    `PhotoType` VARCHAR(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

-- Tabela para definir tipos de usuário
CREATE TABLE `UserType` (
    `UserTypeId` INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `Description` VARCHAR(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

-- Tabela para armazenar informações dos usuários
CREATE TABLE `User` (
    `UserId` INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `Username` VARCHAR(250) NOT NULL,
    `Email` VARCHAR(250) NOT NULL,
    `Password` VARCHAR(100) NOT NULL,
    `UserTypeId` INT(11) NOT NULL,
    `PhotoId` INT(11) NOT NULL,
    FOREIGN KEY (`UserTypeId`) REFERENCES `UserType`(`UserTypeId`),
    FOREIGN KEY (`PhotoId`) REFERENCES `Photograph`(`PhotoId`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

-- Tabela para armazenar classificações dos usuários para corvos
CREATE TABLE `Rank` (
    `RankId` INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    `Date` DATE NOT NULL,
    `UserId` INT(11) NOT NULL,
    `CrowId` INT(11) NOT NULL,
    FOREIGN KEY (`UserId`) REFERENCES `User`(`UserId`),
    FOREIGN KEY (`CrowId`) REFERENCES `Crow`(`CrowId`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
